# Joindre un poste au domaine
Add-Computer -DomainName entreprise.local -Credential (Get-Credential)

# Créer une OU puis un utilisateur

New-ADUser -Name "Sophie Martin" -SamAccountName "rh.sophie" -Path "OU=RessourcesHumaines,DC=entreprise,DC=local" -AccountPassword (ConvertTo-SecureString "Sophie@2024" -AsPlainText -Force) -Enabled $true
